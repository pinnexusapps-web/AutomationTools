Sub GenerateDailyReport()
    ' Software Version 1.4.5
    ' developed by DSU11425

    Dim wsData As Worksheet, wsReport As Worksheet
    Dim lastRow As Long, reportRow As Long, lastReportRow As Long
    Dim pickerDict As Object
    Dim pickerName As Variant
    Dim tableRange As Range
    Dim rawDate As String, formattedDate As String
    Dim summaryRow As Long, modifiedRow As Long
    Dim i As Long, startDataRow As Long
    Dim amtDiff As Double
    Dim cellI As Variant, cellJ As Variant
    Dim numI As Double, numJ As Double

    Dim currentSuccess As Long
    Dim maxTGSuccess As Long, maxTGRow As Long
    Dim maxNonTGSuccess As Long, maxNonTGRow As Long

    ' === LAYOUT CONSTANTS ===
    Const COL_START As Long = 3   ' Starts at C (index 3)
    Const ROW_START As Long = 5   ' row 5

    Set wsData = ActiveSheet

    lastRow = wsData.Cells(wsData.Rows.Count, "A").End(xlUp).Row
    If lastRow <= 1 Then
        MsgBox "The active sheet seems to be empty or has no data rows. Please open the correct file first!", vbCritical, "Error"
        Exit Sub
    End If

    Application.ScreenUpdating = False
    Application.DisplayAlerts = False

    For i = 2 To lastRow
        cellI = wsData.Cells(i, 9).Value
        If Not IsNumeric(cellI) And cellI <> "" Then
            cellI = Replace(cellI, "AED", "")
            cellI = Trim(cellI)
            If IsNumeric(cellI) Then wsData.Cells(i, 9).Value = CDbl(cellI)
        End If
        wsData.Cells(i, 9).NumberFormat = "0.00"

        cellJ = wsData.Cells(i, 10).Value
        If Not IsNumeric(cellJ) And cellJ <> "" Then
            cellJ = Replace(cellJ, "AED", "")
            cellJ = Trim(cellJ)
            If IsNumeric(cellJ) Then wsData.Cells(i, 10).Value = CDbl(cellJ)
        End If
        wsData.Cells(i, 10).NumberFormat = "0.00"
    Next i

    rawDate = wsData.Range("D2").Value
    If InStr(rawDate, " ") > 0 Then
        formattedDate = Split(rawDate, " ")(0)
    Else
        formattedDate = rawDate
    End If

    On Error Resume Next
    Set wsReport = Sheets("Daily Report")
    If Not wsReport Is Nothing Then wsReport.Delete
    On Error GoTo 0

    Set wsReport = Worksheets.Add(Before:=Worksheets(1))
    wsReport.Name = "Daily Report"

    ActiveWindow.DisplayGridlines = False

    With wsReport
        .Columns("A").ColumnWidth = 11
        .Columns("B").ColumnWidth = 1
        .Columns("C").AutoFit
        .Columns("C").ColumnWidth = .Columns("C").ColumnWidth + 3
        .Columns("D").ColumnWidth = 14
        .Columns("E").ColumnWidth = 20
        .Columns("F").ColumnWidth = 14
        .Columns("G").ColumnWidth = 1
        .Columns("H").ColumnWidth = 11
        .Columns("I").ColumnWidth = 12
        
        .Rows("1").RowHeight = 15

        ' =========================================================
        ' PICKER SUMMARY TABLE — starts at C5
        ' =========================================================
        Dim hdrRow As Long
        hdrRow = ROW_START  ' row 5

        .Cells(hdrRow, COL_START).Value = "PICKER"
        .Cells(hdrRow, COL_START + 1).Value = "CANCELLED"
        .Cells(hdrRow, COL_START + 2).Value = "SUCCESS"
        .Cells(hdrRow, COL_START + 3).Value = "PICKED UP"

        With .Range(.Cells(hdrRow, COL_START), .Cells(hdrRow, COL_START + 3))
            .Interior.Color = RGB(214, 227, 242)
            .Font.Bold = True
            .Font.Name = "Calibri"
            .Font.Size = 11
        End With

        Set pickerDict = CreateObject("Scripting.Dictionary")
        For reportRow = 2 To lastRow
            pickerName = wsData.Cells(reportRow, 7).Value
            If pickerName <> "" And pickerName <> "Picker" And pickerName <> "PICKER" _
               And Not pickerDict.Exists(pickerName) Then
                pickerDict.Add pickerName, ""
            End If
        Next reportRow

        reportRow = hdrRow + 1
        For Each pickerName In pickerDict.Keys
            .Cells(reportRow, COL_START).Value = pickerName
            .Cells(reportRow, COL_START + 1).Formula = _
                "=COUNTIFS('" & wsData.Name & "'!G:G,""" & pickerName & """,'" & wsData.Name & "'!C:C,""CANCELLED"")"
            .Cells(reportRow, COL_START + 2).Formula = _
                "=COUNTIFS('" & wsData.Name & "'!G:G,""" & pickerName & """,'" & wsData.Name & "'!C:C,""PICKED_UP"")"
            .Cells(reportRow, COL_START + 3).Formula = _
                "=SUM(" & .Cells(reportRow, COL_START + 1).Address(False, False) & ":" & _
                .Cells(reportRow, COL_START + 2).Address(False, False) & ")"
            reportRow = reportRow + 1
        Next pickerName

        lastReportRow = reportRow - 1

        Set tableRange = .Range(.Cells(hdrRow + 1, COL_START), .Cells(lastReportRow, COL_START + 3))
        .Sort.SortFields.Clear
        .Sort.SortFields.Add _
            Key:=.Range(.Cells(hdrRow + 1, COL_START + 3), .Cells(lastReportRow, COL_START + 3)), _
            SortOn:=xlSortOnValues, Order:=xlDescending, DataOption:=xlSortNormal
        With .Sort
            .SetRange tableRange
            .Header = xlNo
            .MatchCase = False
            .Orientation = xlTopToBottom
            .SortMethod = xlPinYin
            .Apply
        End With

        maxTGSuccess = -1: maxTGRow = 0
        maxNonTGSuccess = -1: maxNonTGRow = 0

        For reportRow = hdrRow + 1 To lastReportRow
            pickerName = .Cells(reportRow, COL_START).Value
            currentSuccess = .Cells(reportRow, COL_START + 2).Value

            If Left(Trim(pickerName), 2) = "TG" Then
                If currentSuccess > maxTGSuccess Then
                    maxTGSuccess = currentSuccess
                    maxTGRow = reportRow
                End If
            Else
                If currentSuccess > maxNonTGSuccess Then
                    maxNonTGSuccess = currentSuccess
                    maxNonTGRow = reportRow
                End If
            End If

            If .Cells(reportRow, COL_START + 1).Value = 0 Then .Cells(reportRow, COL_START + 1).Value = ""
            If .Cells(reportRow, COL_START + 2).Value = 0 Then .Cells(reportRow, COL_START + 2).Value = ""
            .Range(.Cells(reportRow, COL_START), .Cells(reportRow, COL_START + 3)).Font.Name = "Calibri"
            .Range(.Cells(reportRow, COL_START), .Cells(reportRow, COL_START + 3)).Font.Size = 11
        Next reportRow

        If maxTGRow > 0 Then .Range(.Cells(maxTGRow, COL_START), .Cells(maxTGRow, COL_START + 3)).Interior.Color = RGB(204, 229, 255)
        If maxNonTGRow > 0 Then .Range(.Cells(maxNonTGRow, COL_START), .Cells(maxNonTGRow, COL_START + 3)).Interior.Color = RGB(255, 224, 178)

        With .Range(.Cells(hdrRow, COL_START), .Cells(lastReportRow + 1, COL_START + 3))
            .Borders(xlInsideHorizontal).LineStyle = xlContinuous
            .Borders(xlInsideHorizontal).Weight = xlThin
            .Borders(xlInsideHorizontal).Color = RGB(180, 180, 180)
            .Borders(xlInsideVertical).LineStyle = xlContinuous
            .Borders(xlInsideVertical).Weight = xlThin
            .Borders(xlInsideVertical).Color = RGB(180, 180, 180)
            .Borders(xlEdgeLeft).LineStyle = xlContinuous
            .Borders(xlEdgeLeft).Weight = xlThin
            .Borders(xlEdgeLeft).Color = RGB(120, 120, 120)
            .Borders(xlEdgeRight).LineStyle = xlContinuous
            .Borders(xlEdgeRight).Weight = xlThin
            .Borders(xlEdgeRight).Color = RGB(120, 120, 120)
            .Borders(xlEdgeTop).LineStyle = xlContinuous
            .Borders(xlEdgeTop).Weight = xlThin
            .Borders(xlEdgeTop).Color = RGB(120, 120, 120)
            .Borders(xlEdgeBottom).LineStyle = xlContinuous
            .Borders(xlEdgeBottom).Weight = xlThin
            .Borders(xlEdgeBottom).Color = RGB(120, 120, 120)
        End With

        reportRow = lastReportRow + 1
        .Cells(reportRow, COL_START).Value = "Grand Total"
        .Cells(reportRow, COL_START + 1).Formula = "=SUM(" & .Cells(hdrRow + 1, COL_START + 1).Address(False, False) & ":" & .Cells(reportRow - 1, COL_START + 1).Address(False, False) & ")"
        .Cells(reportRow, COL_START + 2).Formula = "=SUM(" & .Cells(hdrRow + 1, COL_START + 2).Address(False, False) & ":" & .Cells(reportRow - 1, COL_START + 2).Address(False, False) & ")"
        .Cells(reportRow, COL_START + 3).Formula = "=SUM(" & .Cells(hdrRow + 1, COL_START + 3).Address(False, False) & ":" & .Cells(reportRow - 1, COL_START + 3).Address(False, False) & ")"

        With .Range(.Cells(reportRow, COL_START), .Cells(reportRow, COL_START + 3))
            .Font.Bold = True
            .Font.Name = "Calibri"
            .Font.Size = 11
            .Interior.Color = RGB(214, 227, 242)
            .Borders(xlEdgeTop).LineStyle = xlContinuous
            .Borders(xlEdgeTop).Weight = xlThin
            .Borders(xlEdgeBottom).LineStyle = xlDouble
        End With

        ' =========================================================
        ' MODIFIED ORDERS TABLE
        ' =========================================================
        modifiedRow = reportRow + 3

        .Cells(modifiedRow, COL_START).Value = "Modified Orders List"
        With .Cells(modifiedRow, COL_START).Font
            .Bold = True
            .Size = 12
            .Name = "Calibri"
        End With

        modifiedRow = modifiedRow + 1
        Dim modHdrRow As Long
        modHdrRow = modifiedRow

        .Cells(modHdrRow, COL_START).Value = "PICKER"           ' C
        .Cells(modHdrRow, COL_START + 1).Value = "ORDER ID"     ' D
        .Range(.Cells(modHdrRow, COL_START + 2), .Cells(modHdrRow, COL_START + 3)).Merge
        .Cells(modHdrRow, COL_START + 2).Value = "REASON"       ' E (merged E:F)
        .Cells(modHdrRow, COL_START + 6).Value = "OVERLAP"      ' I

        Dim colIdx As Variant
        Dim headerCols As Variant
        headerCols = Array(COL_START, COL_START + 1, COL_START + 2, COL_START + 6)
        
        For Each colIdx In headerCols
            With .Cells(modHdrRow, colIdx)
                .Font.Bold = True
                .Font.Name = "Calibri"
                .Font.Size = 11
                .Interior.Color = RGB(230, 240, 250)
                .HorizontalAlignment = xlCenter
                .Borders(xlEdgeTop).LineStyle = xlContinuous
                .Borders(xlEdgeTop).Weight = xlThin
                .Borders(xlEdgeTop).Color = RGB(140, 140, 140)
                .Borders(xlEdgeBottom).LineStyle = xlContinuous
                .Borders(xlEdgeBottom).Weight = xlThin
                .Borders(xlEdgeBottom).Color = RGB(180, 180, 180)
            End With
        Next colIdx

        .Cells(modHdrRow, COL_START).Borders(xlEdgeLeft).LineStyle = xlContinuous
        .Cells(modHdrRow, COL_START).Borders(xlEdgeLeft).Weight = xlThin
        .Cells(modHdrRow, COL_START).Borders(xlEdgeLeft).Color = RGB(140, 140, 140)
        .Cells(modHdrRow, COL_START).Borders(xlEdgeRight).LineStyle = xlContinuous
        .Cells(modHdrRow, COL_START).Borders(xlEdgeRight).Weight = xlThin
        .Cells(modHdrRow, COL_START).Borders(xlEdgeRight).Color = RGB(200, 200, 200)
        .Cells(modHdrRow, COL_START + 1).Borders(xlEdgeRight).LineStyle = xlContinuous
        .Cells(modHdrRow, COL_START + 1).Borders(xlEdgeRight).Weight = xlThin
        .Cells(modHdrRow, COL_START + 1).Borders(xlEdgeRight).Color = RGB(200, 200, 200)
        .Cells(modHdrRow, COL_START + 2).Borders(xlEdgeRight).LineStyle = xlContinuous
        .Cells(modHdrRow, COL_START + 2).Borders(xlEdgeRight).Weight = xlThin
        .Cells(modHdrRow, COL_START + 2).Borders(xlEdgeRight).Color = RGB(200, 200, 200)
        .Cells(modHdrRow, COL_START + 3).Borders(xlEdgeRight).LineStyle = xlContinuous
        .Cells(modHdrRow, COL_START + 3).Borders(xlEdgeRight).Weight = xlThin
        .Cells(modHdrRow, COL_START + 3).Borders(xlEdgeRight).Color = RGB(140, 140, 140)
        .Cells(modHdrRow, COL_START + 6).Borders(xlEdgeRight).LineStyle = xlContinuous
        .Cells(modHdrRow, COL_START + 6).Borders(xlEdgeRight).Weight = xlThin
        .Cells(modHdrRow, COL_START + 6).Borders(xlEdgeRight).Color = RGB(140, 140, 140)

        startDataRow = modHdrRow + 1
        modifiedRow = modHdrRow

        Dim btnTop As Double, btnLeft As Double
        Dim btnWidth As Double, btnHeight As Double
        Dim btn As Object
        Dim btnCell As Range

        For i = 2 To lastRow
            If wsData.Cells(i, 3).Value = "PICKED_UP" Then
                numI = wsData.Cells(i, 9).Value
                numJ = wsData.Cells(i, 10).Value
                amtDiff = Abs(numJ - numI)

                If amtDiff >= 1# Then
                    modifiedRow = modifiedRow + 1

                    With .Cells(modifiedRow, COL_START)
                        .Value = wsData.Cells(i, 7).Value
                        .Font.Name = "Calibri"
                        .Font.Size = 10
                        .HorizontalAlignment = xlCenter
                        .VerticalAlignment = xlCenter
                    End With

                    With .Cells(modifiedRow, COL_START + 1)
                        .Value = "'" & wsData.Cells(i, 1).Value
                        .Font.Name = "Calibri"
                        .Font.Size = 10
                        .HorizontalAlignment = xlCenter
                        .VerticalAlignment = xlCenter
                    End With

                    .Range(.Cells(modifiedRow, COL_START + 2), .Cells(modifiedRow, COL_START + 3)).Merge
                    With .Cells(modifiedRow, COL_START + 2)
                        .Value = "Not Found"
                        .Font.Name = "Calibri"
                        .Font.Size = 10
                        .Font.Bold = True
                        .Font.Color = RGB(180, 0, 0)
                        .HorizontalAlignment = xlCenter
                        .VerticalAlignment = xlCenter
                        .Interior.Color = RGB(255, 235, 235)
                    End With

                    Set btnCell = .Cells(modifiedRow, COL_START + 6)
                    btnTop = btnCell.Top + 2
                    btnLeft = btnCell.Left + 3
                    btnWidth = btnCell.Width - 6
                    btnHeight = btnCell.Height - 4

                    Set btn = wsReport.Buttons.Add(btnLeft, btnTop, btnWidth, btnHeight)
                    With btn
                        .Caption = "Overlap"
                        .Font.Name = "Calibri"
                        .Font.Size = 9
                        .Name = "btnOL_R" & modifiedRow & "C" & (COL_START + 2)
                        .OnAction = "ToggleReasonButton"
                    End With
                    
                    .Range(.Cells(modifiedRow, COL_START), .Cells(modifiedRow, COL_START + 6)).RowHeight = 22
                    
                    .Cells(modifiedRow, COL_START).Borders(xlEdgeLeft).LineStyle = xlContinuous
                    .Cells(modifiedRow, COL_START).Borders(xlEdgeLeft).Weight = xlThin
                    .Cells(modifiedRow, COL_START).Borders(xlEdgeLeft).Color = RGB(140, 140, 140)
                    .Cells(modifiedRow, COL_START).Borders(xlEdgeRight).LineStyle = xlContinuous
                    .Cells(modifiedRow, COL_START).Borders(xlEdgeRight).Weight = xlThin
                    .Cells(modifiedRow, COL_START).Borders(xlEdgeRight).Color = RGB(200, 200, 200)
                    .Cells(modifiedRow, COL_START + 1).Borders(xlEdgeRight).LineStyle = xlContinuous
                    .Cells(modifiedRow, COL_START + 1).Borders(xlEdgeRight).Weight = xlThin
                    .Cells(modifiedRow, COL_START + 1).Borders(xlEdgeRight).Color = RGB(200, 200, 200)
                    .Cells(modifiedRow, COL_START + 2).Borders(xlEdgeRight).LineStyle = xlContinuous
                    .Cells(modifiedRow, COL_START + 2).Borders(xlEdgeRight).Weight = xlThin
                    .Cells(modifiedRow, COL_START + 2).Borders(xlEdgeRight).Color = RGB(200, 200, 200)
                    .Cells(modifiedRow, COL_START + 3).Borders(xlEdgeRight).LineStyle = xlContinuous
                    .Cells(modifiedRow, COL_START + 3).Borders(xlEdgeRight).Weight = xlThin
                    .Cells(modifiedRow, COL_START + 3).Borders(xlEdgeRight).Color = RGB(140, 140, 140)
                    .Cells(modifiedRow, COL_START + 6).Borders(xlEdgeRight).LineStyle = xlContinuous
                    .Cells(modifiedRow, COL_START + 6).Borders(xlEdgeRight).Weight = xlThin
                    .Cells(modifiedRow, COL_START + 6).Borders(xlEdgeRight).Color = RGB(140, 140, 140)
                    
                    Dim dCol As Variant
                    Dim dataCols As Variant
                    dataCols = Array(COL_START, COL_START + 1, COL_START + 2, COL_START + 3, COL_START + 6)
                    For Each dCol In dataCols
                        With .Cells(modifiedRow, dCol)
                            .Borders(xlEdgeBottom).LineStyle = xlContinuous
                            .Borders(xlEdgeBottom).Weight = xlThin
                            .Borders(xlEdgeBottom).Color = RGB(220, 220, 220)
                        End With
                    Next dCol
                End If
            End If
        Next i

        If modifiedRow < startDataRow Then
            .Range(.Cells(startDataRow, COL_START), .Cells(startDataRow, COL_START + 6)).Merge
            .Cells(startDataRow, COL_START).Value = "No modifications recorded today."
            .Cells(startDataRow, COL_START).Font.Italic = True
            modifiedRow = startDataRow
        End If

        ' =========================================================
        ' SUMMARY / WHATSAPP MESSAGE BLOCK
        ' =========================================================
        summaryRow = modifiedRow + 3

        .Cells(summaryRow, COL_START).Value = "Good morning team,"
        
        .Rows(summaryRow).RowHeight = 25

        .Cells(summaryRow + 2, COL_START).Value = "Daily sales report " & formattedDate

        .Cells(summaryRow + 3, COL_START).Value = "Orders:"
        .Cells(summaryRow + 3, COL_START + 1).Formula = "=SUM(" & .Cells(hdrRow + 1, COL_START + 2).Address(False, False) & ":" & .Cells(lastReportRow, COL_START + 2).Address(False, False) & ")"

        .Cells(summaryRow + 4, COL_START).Value = "Cancelled:"
        .Cells(summaryRow + 4, COL_START + 1).Formula = "=SUM(" & .Cells(hdrRow + 1, COL_START + 1).Address(False, False) & ":" & .Cells(lastReportRow, COL_START + 1).Address(False, False) & ")"

        .Cells(summaryRow + 5, COL_START).Value = "Sales :"
        .Cells(summaryRow + 5, COL_START + 1).Formula = "=SUMIFS('" & wsData.Name & "'!J:J,'" & wsData.Name & "'!C:C,""PICKED_UP"")"
        .Cells(summaryRow + 5, COL_START + 1).NumberFormat = "0.00"

        .Cells(summaryRow + 6, COL_START).Value = "Average:"
        .Cells(summaryRow + 6, COL_START + 1).Formula = "=AVERAGEIFS('" & wsData.Name & "'!J:J,'" & wsData.Name & "'!C:C,""PICKED_UP"")"
        .Cells(summaryRow + 6, COL_START + 1).NumberFormat = "0.00"

        Dim modCount As Long
        modCount = 0
        If modifiedRow >= startDataRow Then modCount = modifiedRow - startDataRow + 1

        .Cells(summaryRow + 7, COL_START).Value = "Modification:"
        .Cells(summaryRow + 7, COL_START + 1).Value = modCount
        If modCount > 0 Then
            .Cells(summaryRow + 7, COL_START + 2).Value = "(" & modCount & " Not Found)"
        Else
            .Cells(summaryRow + 7, COL_START + 2).Value = ""
        End If

        With .Cells(summaryRow + 7, COL_START + 2)
            .Font.Name = "Calibri"
            .Font.Size = 11
            .Font.Italic = True
            .Font.Color = RGB(180, 80, 0)
            .HorizontalAlignment = xlLeft
        End With

        On Error Resume Next
        wsReport.Parent.Names("DR_ReasonStart").Delete
        wsReport.Parent.Names("DR_ReasonEnd").Delete
        wsReport.Parent.Names("DR_SummaryLabelCell").Delete
        wsReport.Parent.Names("DR_SummaryRow").Delete
        wsReport.Parent.Names("DR_ModifiedRow").Delete
        wsReport.Parent.Names("DR_StartDataRow").Delete
        wsReport.Parent.Names("DR_ModHdrRow").Delete
        On Error GoTo 0

        If modifiedRow >= startDataRow Then
            wsReport.Parent.Names.Add Name:="DR_ReasonStart", RefersTo:=wsReport.Cells(startDataRow, COL_START + 2)
            wsReport.Parent.Names.Add Name:="DR_ReasonEnd", RefersTo:=wsReport.Cells(modifiedRow, COL_START + 2)
        End If
        wsReport.Parent.Names.Add Name:="DR_SummaryLabelCell", RefersTo:=wsReport.Cells(summaryRow + 7, COL_START + 2)
        
        wsReport.Parent.Names.Add Name:="DR_SummaryRow", RefersTo:=wsReport.Cells(summaryRow, 1)
        wsReport.Parent.Names.Add Name:="DR_ModifiedRow", RefersTo:=wsReport.Cells(modifiedRow, 1)
        wsReport.Parent.Names.Add Name:="DR_StartDataRow", RefersTo:=wsReport.Cells(startDataRow, 1)
        wsReport.Parent.Names.Add Name:="DR_ModHdrRow", RefersTo:=wsReport.Cells(modHdrRow, 1)

        With .Range(.Cells(summaryRow, COL_START), .Cells(summaryRow + 7, COL_START + 1)).Font
            .Name = "Calibri"
            .Size = 11
            .Bold = False
        End With
        .Cells(summaryRow + 2, COL_START).Font.Bold = True

        .Columns("C:F").HorizontalAlignment = xlCenter
        .Range(.Cells(summaryRow, COL_START), .Cells(summaryRow + 7, COL_START)).HorizontalAlignment = xlLeft
        .Range(.Cells(summaryRow + 3, COL_START + 1), .Cells(summaryRow + 7, COL_START + 1)).HorizontalAlignment = xlLeft

        ' =========================================================
        ' PLACING COPY BUTTONS NEXT TO "Good morning team,"
        ' =========================================================
        Dim btnCopyTable As Object, btnCopyText As Object
        Dim cellTableBtn As Range, cellTextBtn As Range
        
        Set cellTableBtn = .Cells(summaryRow, COL_START + 1)
        Set btnCopyTable = wsReport.Buttons.Add(cellTableBtn.Left + 2, cellTableBtn.Top + 2, cellTableBtn.Width - 4, cellTableBtn.Height - 4)
        With btnCopyTable
            .Caption = "Copy Table"
            .Font.Name = "Calibri"
            .Font.Size = 10
            .Font.Bold = True
            .Name = "btnCopyTable"
            .OnAction = "CopyTableData"
        End With
        
        Set cellTextBtn = .Cells(summaryRow, COL_START + 2)
        Set btnCopyText = wsReport.Buttons.Add(cellTextBtn.Left + 2, cellTextBtn.Top + 2, cellTextBtn.Width - 4, cellTextBtn.Height - 4)
        With btnCopyText
            .Caption = "Copy Text"
            .Font.Name = "Calibri"
            .Font.Size = 10
            .Font.Bold = True
            .Name = "btnCopyText"
            .OnAction = "CopyTextData"
        End With

        .Columns("A").ColumnWidth = 11
        .Columns("B").ColumnWidth = 1
        .Columns("C").AutoFit
        .Columns("C").ColumnWidth = .Columns("C").ColumnWidth + 3
        .Columns("D").ColumnWidth = 14
        .Columns("E").ColumnWidth = 20
        .Columns("F").ColumnWidth = 14
        .Columns("G").ColumnWidth = 1
        .Columns("H").ColumnWidth = 11
        .Columns("I").ColumnWidth = 12
    End With

    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
    MsgBox "Daily Report generated successfully!", vbInformation, "Done"
End Sub

' =============================================================
' COPY TABLE DATA - Copies B4 to Gx (x = last row of Mod table + 1)
' =============================================================
Sub CopyTableData()
    Dim ws As Worksheet
    Dim copyRange As Range
    Dim modifiedRow As Long, startDataRow As Long
    
    Set ws = ActiveSheet
    
    On Error Resume Next
    modifiedRow = ws.Parent.Names("DR_ModifiedRow").RefersToRange.Row
    startDataRow = ws.Parent.Names("DR_StartDataRow").RefersToRange.Row
    On Error GoTo 0
    
    If modifiedRow = 0 Then
        MsgBox "Please generate the report first!", vbExclamation, "Error"
        Exit Sub
    End If
    
    Dim lastCopyRow As Long
    If modifiedRow >= startDataRow Then
        lastCopyRow = modifiedRow + 1
    Else
        lastCopyRow = startDataRow
    End If
    
    Set copyRange = ws.Range(ws.Cells(4, 2), ws.Cells(lastCopyRow, 7))
    copyRange.Copy
    
End Sub

' =============================================================
' COPY TEXT DATA - Copies "Good morning team," row to last text row (Col E)
' =============================================================
Sub CopyTextData()
    Dim ws As Worksheet
    Dim summaryRow As Long
    Dim textRange As Range
    
    Set ws = ActiveSheet
    
    On Error Resume Next
    summaryRow = ws.Parent.Names("DR_SummaryRow").RefersToRange.Row
    On Error GoTo 0
    
    If summaryRow = 0 Then
        MsgBox "Please generate the report first!", vbExclamation, "Error"
        Exit Sub
    End If
    
    Set textRange = ws.Range(ws.Cells(summaryRow, 3), ws.Cells(summaryRow + 7, 5))
    textRange.Copy
    
End Sub

' =============================================================
' TOGGLE HANDLER — called by every Overlap button click
' =============================================================
Sub ToggleReasonButton()
    Dim ws As Worksheet
    Dim btnName As String
    Dim parsed As String
    Dim cPos As Long
    Dim targetRow As Long, targetCol As Long
    Dim reasonCell As Range

    Set ws = ActiveSheet
    btnName = Application.Caller

    parsed = Mid(btnName, Len("btnOL_R") + 1)
    cPos = InStr(parsed, "C")
    targetRow = CLng(Left(parsed, cPos - 1))
    targetCol = CLng(Mid(parsed, cPos + 1))

    Set reasonCell = ws.Cells(targetRow, targetCol)

    If reasonCell.Value = "Not Found" Then
        With reasonCell
            .Value = "Overlap"
            .Font.Color = RGB(180, 80, 0)
            .Interior.Color = RGB(255, 243, 205)
        End With
    Else
        With reasonCell
            .Value = "Not Found"
            .Font.Color = RGB(180, 0, 0)
            .Interior.Color = RGB(255, 235, 235)
        End With
    End If

    Call UpdateModificationSummary(ws)
End Sub

' =============================================================
' Recalculates the overlap/notfound summary label
' =============================================================
Sub UpdateModificationSummary(ws As Worksheet)
    Dim reasonStart As Range, reasonEnd As Range, labelCell As Range
    Dim reasonRange As Range
    Dim c As Range
    Dim overlapCount As Long, notFoundCount As Long, total As Long
    Dim labelText As String

    On Error Resume Next
    Set reasonStart = ws.Parent.Names("DR_ReasonStart").RefersToRange
    Set reasonEnd = ws.Parent.Names("DR_ReasonEnd").RefersToRange
    Set labelCell = ws.Parent.Names("DR_SummaryLabelCell").RefersToRange
    On Error GoTo 0

    If reasonStart Is Nothing Or reasonEnd Is Nothing Or labelCell Is Nothing Then Exit Sub

    Set reasonRange = ws.Range(reasonStart, reasonEnd)

    overlapCount = 0
    notFoundCount = 0
    total = 0

    For Each c In reasonRange
        If c.MergeCells Then
            If c.Address = c.MergeArea.Cells(1, 1).Address Then
                total = total + 1
                Select Case Trim(c.Value)
                    Case "Overlap":   overlapCount = overlapCount + 1
                    Case "Not Found": notFoundCount = notFoundCount + 1
                End Select
            End If
        Else
            total = total + 1
            Select Case Trim(c.Value)
                Case "Overlap":   overlapCount = overlapCount + 1
                Case "Not Found": notFoundCount = notFoundCount + 1
            End Select
        End If
    Next c

    If total = 0 Then
        labelText = ""
    ElseIf overlapCount > 0 Then
        labelText = "(" & overlapCount & " Overlap" & IIf(overlapCount > 1, "s", "") & ")"
    ElseIf notFoundCount > 0 Then
        labelText = "(" & notFoundCount & " Not Found)"
    Else
        labelText = ""
    End If

    labelCell.Value = labelText
End Sub
