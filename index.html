PDF Auto Rename v1.0

OCR scans every PDF in a folder · extracts CN / PO number · renames automatically

rename_docs.py AutoRename.bat 
# rename_docs.py  —  PDF Auto Rename v1.0
import os, sys, re, fitz, easyocr, numpy as np, cv2

reader = easyocr.Reader(['en'])
CN_PATTERN = r"(CN-[A-Za-z0-9]+-[0-9]+)"
PO_PATTERN = r"(PO[0-9]+)"

def extract_id_from_pdf(pdf_path):
    doc  = fitz.open(pdf_path)
    page = doc[0]
    pix  = page.get_pixmap(matrix=fitz.Matrix(2, 2))
    img  = np.frombuffer(pix.samples, dtype=np.uint8).reshape((pix.h, pix.w, pix.n))
    text = " ".join(easyocr.Reader(['en']).readtext(
               cv2.cvtColor(img, cv2.COLOR_BGR2RGB), detail=0)).upper()
    m = re.search(CN_PATTERN, text) or re.search(PO_PATTERN, text)
    return m.group(1).replace(" ", "") if m else None

def process_and_rename_pdfs(folder):
    for f in [x for x in os.listdir(folder) if x.lower().endswith('.pdf')]:
        path   = os.path.join(folder, f)
        doc_id = extract_id_from_pdf(path)
        if doc_id:
            new_path = os.path.join(folder, f"{doc_id}.pdf")
            if not os.path.exists(new_path):
                os.rename(path, new_path)
                print(f"--> SUCCESS: {f}  →  {doc_id}.pdf")
            else:
                print(f"--> SKIPPED: {doc_id}.pdf already exists")
        else:
            print(f"--> FAILED : no CN/PO pattern found in {f}")

process_and_rename_pdfs(sys.argv[1] if len(sys.argv) > 1 else "./")
